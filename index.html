📋 ประวัติการสร้าง Label
📌 วิธีตั้งค่า Google Apps Script เพื่อบันทึกข้อมูลลง Sheet
ขั้นตอน:
1. เปิด Google Sheet → เมนู Extensions → Apps Script
2. วางโค้ดด้านล่าง → บันทึก → Deploy → New Deployment → Web App
3. ตั้ง Execute as: Me / Who has access: Anyone
4. Copy URL ที่ได้ไปวางใน input "Apps Script URL" ด้านล่าง

โค้ด Apps Script (วางทั้งหมด): 
function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var ss   = SpreadsheetApp.openById(data.sheetId);
    var type = data.type; // "RM" or "FG"

    if (type === "RM") {
      var ws = ss.getSheetByName("RawMaterial_Log");
      var lastRow = ws.getLastRow();
      var nextNum = lastRow; // row count = IMT number
      var imtNo   = "IMT-" + String(nextNum).padStart(5,"0");
      var now     = new Date();
      ws.appendRow([
        imtNo,                        // IMT No.
        data.date,                    // วันที่
        data.shift || "Day",          // Shift
        data.location || "Store",     // Form Location
        data.partNo,                  // Part No.
        data.materialName,            // วัสดุ
        data.size || "",              // ขนาด
        data.rollNo,                  // Roll No.
        data.lot,                     // Lot
        parseFloat(data.qty),         // จำนวน
        data.unit,                    // หน่วย
        data.mc || "",                // M/C
        data.refNo || "",             // Ref No. R/C
        data.poNo || "",              // PO No.
        data.supplier || "",          // Invoice Supplier
        data.remark || "",            // หมายเหตุ
        data.employeeName || "",      // ชื่อพนักงาน
        data.employeeId || "",        // E/N
        now                           // บันทึกเมื่อ
      ]);
      return ContentService.createTextOutput(JSON.stringify({
        ok: true, imtNo: imtNo
      })).setMimeType(ContentService.MimeType.JSON);
    }

    if (type === "FG") {
      var ws = ss.getSheetByName("FG_Log");
      var lastRow = ws.getLastRow();
      var nextNum = lastRow;
      var fgNo    = "FG-" + String(nextNum).padStart(5,"0");
      var now     = new Date();
      ws.appendRow([
        fgNo,                         // FG No.
        data.date,                    // วันที่
        data.partNo,                  // รหัสสินค้า
        data.materialName,            // ชื่อสินค้า
        parseFloat(data.qty),         // จำนวน
        data.unit,                    // หน่วย
        data.lot,                     // Lot
        data.mc || "",                // M/C
        data.poNo || "",              // PO Ref
        data.shift || "Day",          // Shift
        data.employeeName || "",      // ชื่อพนักงาน
        data.employeeId || "",        // E/N
        now                           // บันทึกเมื่อ
      ]);
      return ContentService.createTextOutput(JSON.stringify({
        ok: true, fgNo: fgNo
      })).setMimeType(ContentService.MimeType.JSON);
    }

    return ContentService.createTextOutput(JSON.stringify({ok:false, msg:"Unknown type"}))
      .setMimeType(ContentService.MimeType.JSON);

  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({ok:false, msg:err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
